    }
    highlightedArticles: allArticlesYaml(
      limit: 3

      sort: { fields: [published], order: DESC }

          title
          url
          outlet_name
          formattedDate: published(formatString: "MMMM DD, YYYY")
        }
      }
    }
    recentArticles: allArticlesYaml(limit: 3, sort: { fields: [published], order: DESC }) {
      list: edges {

      }
    }
